CREATE TABLE [dbo].[Order] (
    [Id]         INT           IDENTITY (1, 1) NOT NULL,
    [CustomerId] INT           NOT NULL,
    [OrderDate]  DATETIME2 (7) NULL,
    [Status]     INT           NULL,
    PRIMARY KEY CLUSTERED ([Id] ASC),
    CONSTRAINT [FK_CustomerID] FOREIGN KEY ([CustomerId]) REFERENCES [dbo].[User] ([Id])
);

CREATE TABLE [dbo].[OrderProduct] (
    [ProductId] INT             NOT NULL,
    [OrderId]   INT             NOT NULL,
    [Price]     NUMERIC (10, 2) NULL,
    [Quantity]  INT             NULL,
    CONSTRAINT [FK_OrderId] FOREIGN KEY ([OrderId]) REFERENCES [dbo].[Order] ([Id]),
    CONSTRAINT [FK_ProductId] FOREIGN KEY ([ProductId]) REFERENCES [dbo].[Product] ([Id])
);

CREATE TABLE [dbo].[Product] (
    [Id]        INT             IDENTITY (1, 1) NOT NULL,
    [Name]      VARCHAR (64)    NULL,
    [Price]     NUMERIC (10, 2) NULL,
    [Quantity]  INT             NULL,
    [IsDeleted] BIT             NULL,
    [Category]  INT             NULL,
    PRIMARY KEY CLUSTERED ([Id] ASC)
);

CREATE TABLE [dbo].[User] (
    [Id]       INT          IDENTITY (1, 1) NOT NULL,
    [Name]     VARCHAR (50) NULL,
    [Password] VARCHAR (50) NULL,
    [Role]     INT          NULL,
    PRIMARY KEY CLUSTERED ([Id] ASC)
);

CREATE TABLE [dbo].[UserRole] (
    [Id]   INT          IDENTITY (1, 1) NOT NULL,
    [Role] VARCHAR (32) NULL,
    PRIMARY KEY CLUSTERED ([Id] ASC)
);
